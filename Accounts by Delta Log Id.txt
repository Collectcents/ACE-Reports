Select
account.debt_id,
account.client_account_number,
account.customer_client_code,
account.current_balance,
account.debt_status_code,
account.last_status_change_date,
account.score1 as "EQ Score",
worklist.label as "Worklist",
auth_user_passwd.username as "Primary Agent"

From account

LEFT JOIN worklist On account.worklist_id=worklist.worklist_id
LEFT JOIN auth_user_passwd On account.primary_collector_id=auth_user_passwd.user_id

Where account.delta_log_id In ('197701','196234','196235')
