WITH numbered_statuses AS (
    SELECT
        debt.debt_id,
        debt.reference_number,
        debt.customer_client_code,
        (SELECT client.company_name FROM client WHERE client.client_id = debt.client_id) as company_name,
        ROW_NUMBER() OVER (PARTITION BY debt.customer_client_code ORDER BY debt.create_date) as row_num
    FROM debt
    WHERE lower(coalesce(debt.customer_client_code,'')) IN (lower('CM0131C'),lower('CM0155R'),lower('CM0156'),lower('JD4433R'),lower('JK1103R'),lower('JK1016R'),lower('RC1926C'),lower('JK1110'),lower('JK1047'),lower('JK1059'),lower('JK1001'),lower('RC1955R'),lower('JK0906R'),lower('RC1938C'),lower('JK1071R'),lower('CG0037C'),lower('CM0090C'),lower('CM0099C'),lower('JK0918'),lower('JK0486R'),lower('CM0109R'),lower('JK1194C'),lower('JK1100C'),lower('JK1118C'),lower('JK1088C'),lower('JK1129C'),lower('JK1136'),lower('JF1029R'),lower('JK1188C'),lower('JK0613C'),lower('JK1060R'),lower('RC1940'),lower('CM0080R'),lower('CM0142R'),lower('JK0719R'))
)

SELECT debt_id,reference_number,customer_client_code,company_name FROM numbered_statuses WHERE row_num <= 5;
