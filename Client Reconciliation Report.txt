debt.debt_id,
CASE WHEN debt.reference_number IS NUll Then debt.debt_id
ELSE debt.reference_number
END,
debt.client_account_number,
debt.customer_client_code,
debt.client_name,

debt.interest_assigned,
debt.fee_assigned,

debt.last_payment_amount,
